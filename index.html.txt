        function toggleLanguage() {
            isArabic = !isArabic;
            const translateBtn = document.getElementById('translateBtn');
            
            if (isArabic) {
                translateBtn.innerHTML = '🇬🇧 English';
                translateToArabic();
                document.body.style.direction = 'rtl';
                document.body.style.fontFamily = 'Arial, sans-serif';
            } else {
                translateBtn.innerHTML = '🇦🇪 العربية';
                translateToEnglish();
                document.body.style.direction = 'ltr';
                document.body.style.fontFamily = "'Segoe UI', Tahoma, Geneva, Verdana, sans-serif";
            }
        }
        
        function translateToArabic() {
            // Get all text elements and translate them
            const elementsToTranslate = document.querySelectorAll('h1, h2, h3, p, a, button, .stat-label, .player-details, .gift-label, .award, .special-message');
            
            elementsToTranslate.forEach(element => {
                const originalText = element.textContent.trim();
                if (translations[originalText]) {
                    element.setAttribute('data-original', originalText);
                    element.textContent = translations[originalText];
                }
                
                // Handle partial matches for complex text
                let translatedText = originalText;
                Object.keys(translations).forEach(key => {
                    if (originalText.includes(key)) {
                        translatedText = translatedText.replace(key, translations[key]);
                    }
                });
                
                if (translatedText !== originalText) {
                    element.setAttribute('data-original', originalText);
                    element.textContent = translatedText;
                }
            });
            
            // Special handling for specific elements
            const heroTitle = document.querySelector('.hero h1');
            if (heroTitle) {
                heroTitle.setAttribute('data-original', heroTitle.textContent);
                heroTitle.innerHTML = '🌟 رونالدو القادم هنا! 🌟';
            }
            
            const playerName = document.querySelector('.player-name');
            if (playerName) {
                playerName.setAttribute('data-original', playerName.textContent);
                playerName.textContent = 'حمد سعيد الجنيبي';
            }
        }
        
        function translateToEnglish() {
            // Restore original English text
            const translatedElements = document.querySelectorAll('[data-original]');
            translatedElements.forEach(element => {
                element.textContent = element.getAttribute('data-original');
                element.removeAttribute('data-original');
            });
        }
