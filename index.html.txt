        function translateToArabic() {
            // Get all text elements and translate them
            const elementsToTranslate = document.querySelectorAll('h1, h2, h3, p, a, button, .stat-label, .player-details, .gift-label, .award, .special-message');
            
            elementsToTranslate.forEach(element => {
                const originalText = element.textContent.trim();
                if (translations[originalText]) {
                    element.setAttribute('data-original', originalText);
                    element.textContent = translations[originalText];
                }
                
                // Handle partial matches for complex text
                let translatedText = originalText;
                Object.keys(translations).forEach(key => {
                    if (originalText.includes(key)) {
                        translatedText = translatedText.replace(key, translations[key]);
                    }
                });
                
                if (translatedText !== originalText) {
                    element.setAttribute('data-original', originalText);
                    element.textContent = translatedText;
                }
            });
            
            // Special handling for specific elements
            const heroTitle = document.querySelector('.hero h1');
            if (heroTitle) {
                heroTitle.setAttribute('data-original', heroTitle.textContent);
